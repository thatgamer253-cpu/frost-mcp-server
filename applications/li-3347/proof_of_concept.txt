```python
import requests
from bs4 import BeautifulSoup

def scrape_linkedin_jobs(keyword, location):
    """
    A function to scrape LinkedIn for job postings based on a keyword and location.
    This is a basic scraper using requests and BeautifulSoup.
    """
    base_url = 'https://www.linkedin.com/jobs/search/'
    params = {
        'keywords': keyword,
        'location': location,
        'trk': 'public_jobs_jobs-search-bar_search-submit'
    }
    
    response = requests.get(base_url, params=params)
    if response.status_code == 200:
        soup = BeautifulSoup(response.text, 'html.parser')
        jobs = []
        
        # Iterate over the job postings found in the page's HTML content
        for job_anchor in soup.find_all('a', class_='result-card__full-card-link'):
            job_title = job_anchor.text.strip()
            job_link = job_anchor['href']

            # Capture more job details (if needed, such as company name)
            company = job_anchor.find_next('h4', class_='result-card__subtitle').text.strip()

            jobs.append({'title': job_title, 'company': company, 'link': job_link})

        return jobs
    else:
        # Handle potential errors, e.g., if the page could not be fetched
        raise Exception('Failed to retrieve jobs from LinkedIn.')

# Usage of the scrape_linkedin_jobs function
jobs = scrape_linkedin_jobs('AI Developer', 'Remote')

for job in jobs:
    print(f"Job Title: {job['title']}, Company: {job['company']}, Link: {job['link']}")
```
