```python
# PoC for Applied AI Engineer (Python/Scraping/AI Job)
# This script uses requests and BeautifulSoup to scrape headlines from a news website
# and utilizes a simple AI model to perform sentiment analysis on the headlines.

import requests
from bs4 import BeautifulSoup
from textblob import TextBlob

def get_news_headlines(url):
    # Send a GET request to the provided URL
    response = requests.get(url)
    response.raise_for_status()  # Raise an error for unsuccessful requests

    # Create a BeautifulSoup object and parse the page
    soup = BeautifulSoup(response.text, 'html.parser')

    # Extract all news headlines
    headlines = [headline.get_text() for headline in soup.find_all('h3')]

    return headlines

def perform_sentiment_analysis(headlines):
    # Perform sentiment analysis on each headline
    sentiments = {
        "headline": [],
        "sentiment": []
    }

    for headline in headlines:
        # Analyze sentiment using TextBlob
        analysis = TextBlob(headline)
        sentiment = analysis.sentiment.polarity  # Polarity ranges from -1 (negative) to 1 (positive)

        sentiments["headline"].append(headline)
        sentiments["sentiment"].append(sentiment)

    return sentiments

if __name__ == '__main__':
    # URL of the website to scrape
    news_url = 'https://news.ycombinator.com/'  # Example URL

    # Get news headlines
    headlines = get_news_headlines(news_url)

    # Perform sentiment analysis on the extracted headlines
    sentiment_results = perform_sentiment_analysis(headlines)

    # Output the results
    for headline, sentiment in zip(sentiment_results['headline'], sentiment_results['sentiment']):
        print(f"Headline: {headline} | Sentiment: {'Positive' if sentiment > 0 else 'Negative' if sentiment < 0 else 'Neutral'}")
```
