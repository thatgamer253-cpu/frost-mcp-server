```python
# Python script template for web scraping automation

import requests
from bs4 import BeautifulSoup

# Define the URL to scrape
url = 'https://www.example.com'

# Function to handle the web scraping
def scrape_website(url):
    # Send a GET request to the specified URL
    response = requests.get(url)

    # Check if the request was successful
    if response.status_code == 200:
        # Parse the HTML content using BeautifulSoup
        soup = BeautifulSoup(response.content, 'html.parser')

        # Extract required data. Example: all text within <h2> tags
        headlines = soup.find_all('h2')
        for headline in headlines:
            print(headline.text.strip())
    else:
        print(f"Failed to retrieve content: Status {response.status_code}")

# Execute the function with the specified URL
if __name__ == "__main__":
    scrape_website(url)

# Comments:
# 1. This script sets up basic web scraping functionality using Python.
# 2. Utilizes 'requests' for HTTP requests to the target URL.
# 3. 'BeautifulSoup' is used to parse HTML and extract desired data, in this case, all <h2> tag contents.
# 4. Replace the example URL and extraction logic with those relevant to your needs.
```
