# Proof of Concept: Python Script for LinkedIn Data Scraping using an AI Model

import requests
from bs4 import BeautifulSoup
import openai

# Function to send a request to LinkedIn and fetch the HTML content
def fetch_linkedin_page(url):
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36',
        'cookie': '<Your_LinkedIn_Cookies_Here>'
    }
    response = requests.get(url, headers=headers)
    if response.ok:
        return response.text
    else:
        response.raise_for_status()

# Function to parse LinkedIn profile information using BeautifulSoup
def parse_linkedin_profile(html_content):
    soup = BeautifulSoup(html_content, 'html.parser')
    profile_data = {}

    # Extract profile name
    name_tag = soup.find('h1')
    if name_tag:
        profile_data['name'] = name_tag.get_text(strip=True)

    # Extract current position
    position_tag = soup.find('div', class_='current-position')
    if position_tag:
        profile_data['current_position'] = position_tag.get_text(strip=True)

    # Extract education information
    education_section = soup.find('section', {'id': 'education-section'})
    if education_section:
        education_list = education_section.find_all('h3')
        profile_data['education'] = [ed.get_text(strip=True) for ed in education_list]

    return profile_data

# Function to summarize profile data using OpenAI GPT model
def summarize_profile(profile_data, openai_api_key):
    openai.api_key = openai_api_key
    response = openai.Completion.create(
      engine="text-davinci-003",
      prompt=f"Summarize the following LinkedIn profile info: {profile_data}",
      max_tokens=50
    )
    return response.choices[0].text.strip()

# Sample Usage
if __name__ == "__main__":
    linkedin_url = 'https://www.linkedin.com/in/sampleprofile/'
    openai_api_key = 'your-openai-api-key-here'

    try:
        html_content = fetch_linkedin_page(linkedin_url)  # Fetch LinkedIn HTML content
        profile_data = parse_linkedin_profile(html_content)  # Parse the profile data
        summary = summarize_profile(profile_data, openai_api_key)  # Summarize profile data
        print("Profile Summary:", summary)
    except Exception as e:
        print("An error occurred:", e)
