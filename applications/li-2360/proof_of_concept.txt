```python
# Proof of Concept for a Python/Scraping/AI job
# This functional Python script uses BeautifulSoup for web scraping and simple AI task using OpenAI's GPT model

import requests
from bs4 import BeautifulSoup
import openai

# Initialize OpenAI API
openai.api_key = 'YOUR_OPENAI_API_KEY'

def fetch_web_content(url):
    # Make HTTP GET request to fetch the raw HTML content
    response = requests.get(url)
    if response.status_code == 200:
        return response.text
    else:
        raise Exception(f"Failed to fetch web content. Status Code: {response.status_code}")

def extract_data_from_html(html_content):
    # Parse the HTML content with BeautifulSoup
    soup = BeautifulSoup(html_content, 'html.parser')
    # Extract data based on tags or css selectors; placeholder for actual logic
    titles = [title.get_text() for title in soup.find_all('h1')]
    return titles

def generate_summary(data):
    # Simple AI task: Summarize extracted data using OpenAI's GPT model
    response = openai.Completion.create(
        engine="text-davinci-003",
        prompt="Summarize the following text: {}".format(data),
        temperature=0.5,
        max_tokens=50
    )
    summary = response.choices[0].text.strip()
    return summary

# Example usage
if __name__ == "__main__":
    # URL of the page to scrape
    url = 'https://www.example.com'
    
    try:
        # Fetch and parse web content
        html_content = fetch_web_content(url)
        extracted_data = extract_data_from_html(html_content)
        
        # Perform AI task on the extracted data
        summary = generate_summary(" ".join(extracted_data))
        
        # Output summary
        print("Extracted Titles Summary:")
        print(summary)
    except Exception as e:
        print(f"Error: {e}")
```
